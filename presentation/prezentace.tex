\begin{frame}\frametitle{Classification: postprocessing}
    \begin{itemize}
        \item For \emph{localization} cluster analysis is used.
            \begin{itemize}
                \item K-means
                \item Medoids (PAM)
            \end{itemize}
        \item \emph{Count of people} by minimal within-cluster sum of squares.
    \end{itemize}
\end{frame}
